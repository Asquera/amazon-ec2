 42   # Builds the canonical string for signing. This strips out all '&', '?', and '='
 43   # from the query string to be signed.
 44   #   Note:  The parameters in the path passed in must already be sorted in
 45   #   case-insensitive alphabetical order and must not be url encoded.
 46   def EC2.canonical_string(path)
 47     buf = ""
 48     path.split('&').each { |field|
 49       buf << field.gsub(/\&|\?/,"").sub(/=/,"")
 50     }
 51     return buf
 52   end

 54   # Encodes the given string with the secret_access_key, by taking the
 55   # hmac-sha1 sum, and then base64 encoding it.  Optionally, it will also
 56   # url encode the result of that to protect the string if it's going to
 57   # be used as a query string parameter.
 58   def EC2.encode(secret_access_key, str, urlencode=true)
 59     digest = OpenSSL::Digest::Digest.new('sha1')
 60     b64_hmac =
 61       Base64.encode64(
 62         OpenSSL::HMAC.digest(digest, secret_access_key, str)).strip
 63         
 64     if urlencode
 65       return CGI::escape(b64_hmac)
 66     else
 67       return b64_hmac
 68     end
